def leaves
  subquery = select("count(DISTINCT subquery.#{ltree_path_column}) = 1").to_sql
  subquery << "AS subquery WHERE subquery.#{ltree_path_column} <@ #{table_name}.#{ltree_path_column}"
  where subquery
end
